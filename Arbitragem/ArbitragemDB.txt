CREATE DATABASE ArbitragemDB;
USE ArbitragemDB;

CREATE TABLE Arbitros (
    id_arbitro* INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100) NOT NULL,
    ranking ENUM('bronze', 'prata', 'ouro') NOT NULL,
    experiencia INT NOT NULL, -- Experiência em número de jogos arbitrados
    especialidade VARCHAR(50) -- Tipo de jogo em que o árbitro se destaca
    jogos id_jogo*
);

CREATE TABLE Jogos (
    id_jogo* INT PRIMARY KEY AUTO_INCREMENT,
    data_jogo VARCHAR(10) NOT NULL,
    tipo ENUM('bronze', 'prata', 'ouro') NOT NULL,
    endereco: VARCHAR(255),
    descricao VARCHAR(255)
);
